    <script>

        // Menu prices

        const prices = {

            jerkChicken: 15.99,

            curryGoat: 18.99,

            brownStewChicken: 14.99,

            escovitchFish: 16.99,

            oxtail: 22.99,

            ackeeSaltfish: 17.99,

            festival: 4.99,

            plantains: 5.99,

            sorrel: 3.99,

            gingerBeer: 3.99

        };


        const deliveryFee = 3.99;


        function updateQuantity(item, change) {

            const input = document.getElementById(item);

            const currentValue = parseInt(input.value) || 0;

            const newValue = Math.max(0, currentValue + change);

            input.value = newValue;

            updateTotal();

        }


        function updateTotal() {

            let subtotal = 0;

            

            for (const [item, price] of Object.entries(prices)) {

                const quantity = parseInt(document.getElementById(item).value) || 0;

                subtotal += quantity * price;

            }

            

            const total = subtotal + deliveryFee;

            

            document.getElementById('totalAmount').textContent = `$${subtotal.toFixed(2)}`;

            document.getElementById('finalTotal').textContent = `$${total.toFixed(2)}`;

        }


        // Handle form submission

        document.getElementById('orderForm').addEventListener('submit', function(e) {

            // Check if items are selected

            let hasItems = false;

            for (const [item, price] of Object.entries(prices)) {

                const quantity = parseInt(document.getElementById(item).value) || 0;

                if (quantity > 0) {

                    hasItems = true;

                    break;

                }

            }

            

            if (!hasItems) {

                e.preventDefault();

                alert('Please select at least one item before placing your order.');

                return;

            }

            

            // Create hidden fields with order summary for email

            const form = this;

            

            // Remove any existing summary fields

            const existingSummary = form.querySelector('input[name="orderSummary"]');

            if (existingSummary) existingSummary.remove();

            

            // Create order summary

            let orderText = "ORDER SUMMARY:\n\n";

            let subtotal = 0;

            

            for (const [item, price] of Object.entries(prices)) {

                const quantity = parseInt(document.getElementById(item).value) || 0;

                if (quantity > 0) {

                    const itemTotal = quantity * price;

                    subtotal += itemTotal;

                    const itemName = item.replace(/([A-Z])/g, ' $1').replace(/^./, str => str.toUpperCase());

                    orderText += `${quantity}x ${itemName} - ${itemTotal.toFixed(2)}\n`;

                }

            }

            

            orderText += `\nSubtotal: ${subtotal.toFixed(2)}`;

            orderText += `\nDelivery Fee: ${deliveryFee.toFixed(2)}`;

            orderText += `\nTOTAL: ${(subtotal + deliveryFee).toFixed(2)}`;

            

            // Add hidden field with complete order summary

            const summaryField = document.createElement('input');

            summaryField.type = 'hidden';

            summaryField.name = 'orderSummary';

            summaryField.value = orderText;

            form.appendChild(summaryField);

            

            // Add subject line for email

            const subjectField = document.createElement('input');

            subjectField.type = 'hidden';

            subjectField.name = '_subject';

            subjectField.value = `New Order from ${document.getElementById('customerName').value}`;

            form.appendChild(subjectField);

        });
